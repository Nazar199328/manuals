Некоторые команды sed и gawk.
-----------------------------

sed -n '/first/p' file.txt	    # Печать строчки, содержащей шаблон
# '^' - начало строки; '$'- конец строки
sed -n '/^first/p' file.txt	    # Печать строчки, содержащей шаблон в начале строки
sed -n '/first$/p' file.txt	    # Печать строчки, содержащей шаблон в конце строки
sed -n '/[ch]at/p' file.txt     # Класс символов: либо cat, либо hat
sed -n '/[^ch]at/p' file.txt    # Обращение класса символов: любой символ, кроме 'c' и 'h'
sed -n '/.at/p' file.txt        # '.' - означает любой символ
sed -n '/ca*t/p' file.txt       # Символ 'a' встречается нуль или более раз
sed -n '/ca?t/p' file.txt       # Символ 'a' встречается нуль или один раз

sed -n '$p' file.txt 		    # Печать последней строки в файле

# Ключ -n заставляет sed вывести только строки, совпадающие с указанным шаблоном.

sed '2d' file.txt               # Удаление 2-ей строчки
sed '/first/d' file.txt 	    # Удаление строчки по шаблону
sed '/^$/d' file.txt 		    # Удалить все пустые строки

sed 's/first/ooo/' file.txt 	# Замена по шаблону
sed 's/first/ooo/2' file.txt 	# Замена по шаблону второго вхождения
sed 's/first/ooo/g' file.txt 	# Замена по шаблону всех вхождений в строке
sed 's/^first/ooo/g' file.txt 	# Замена если шаблон в начале строки
sed 's/first$/ooo/g' file.txt 	# Замена если шаблон в конце строки

sed 'команда1; команда2'    	    # Oбъединение команд
sed -e 'команда1; команда2'    	    # Oбъединение команд
sed ':a;N;$!ba;s/\n//g' file.txt 	# Oбъединение строк
sed 's/<[^>]*>//g'                  # Удаление дескрипторов HTML
# Конструкция вида: сначала '<'
# потом [^>]* - любой символ кроме '>' встречающийся нуль или более раз
# затем '>' будет заменена на пустой символ, т.е. удалена, во всей строке

sed '=' file.txt			# Oтображает номера строк на отдельной строке
sed '=' file.txt | sed 'N; s/\n/ /' 	# Отображает номера строк на одной строке с текстом

grep -i 'first' file.txt 	# Печать строчки, содержащей шаблон без учета регистра
grep -c 'first' file.txt 	# Вывести число строк, содержащих шаблон
grep -o 'template' file.txt	# Печатает часть строки, соответствущую шаблону template 
grep -l 'first' folder 		# Cписок файлов из папки folder, в которых содержится шаблон
grep -C 1 'template' file.txt	# Печатает строки в окрестности строки с шаблоном
grep -A 1 'template' file.txt	# Печатает строки ниже строки с шаблоном
grep -B 1 'template' file.txt	# Печатает строки выше строки с шаблоном
